// Nama-nama Sheet/Tab yang dibutuhkan aplikasi const SHEET_DATA = 'Akun Sosial Media'; const SHEET_USERS = 'Pengguna App'; // FUNGSI PINTAR: Mengecek dan membuat sheet secara otomatis function getOrCreateSheet(ss, sheetName, headers, columnWidths, defaultRowData = null) { let sheet = ss.getSheetByName(sheetName); if (!sheet) { sheet = ss.insertSheet(sheetName); sheet.appendRow(headers); sheet.getRange(1, 1, 1, headers.length) .setFontWeight('bold') .setBackground('#f3f4f6') .setFontColor('#1e293b'); if (columnWidths && columnWidths.length > 0) { columnWidths.forEach((width, index) => { sheet.setColumnWidth(index + 1, width); }); } if (defaultRowData) { sheet.appendRow(defaultRowData); } } return sheet; } // Inisialisasi Database Dinamis function initDatabase() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheetData = getOrCreateSheet( ss, SHEET_DATA, ['Timestamp', 'Jenis Akun', 'Nama Pemilik', 'Email', 'Kata Sandi'], [150, 150, 150, 200, 150] ); const sheetUsers = getOrCreateSheet( ss, SHEET_USERS, ['Username', 'Password'], [200, 200], ['admin', 'admin123'] ); return { sheetData, sheetUsers }; } // API POST function doPost(e) { try { const db = initDatabase(); const data = JSON.parse(e.postData.contents); const action = data.action; // --- PROSES LOGIN --- if (action === 'login') { const usersData = db.sheetUsers.getDataRange().getValues(); let isAuthenticated = false; for (let i = 1; i < usersData.length; i++) { if (usersData[i][0] === data.username && usersData[i][1] === data.password) { isAuthenticated = true; break; } } if (isAuthenticated) return responseJSON('success', 'Login berhasil!'); else return responseJSON('error', 'Username atau Password salah!'); } // --- PROSES GANTI USERNAME & PASSWORD LOGIN --- else if (action === 'change_auth') { const usersData = db.sheetUsers.getDataRange().getValues(); let userRowIndex = -1; // Cari baris user lama for (let i = 1; i < usersData.length; i++) { if (usersData[i][0] === data.oldUsername) { userRowIndex = i + 1; // +1 karena array index mulai dari 0, baris sheet dari 1 break; } } if (userRowIndex !== -1) { db.sheetUsers.getRange(userRowIndex, 1, 1, 2).setValues([[ data.newUsername, data.newPassword ]]); return responseJSON('success', 'Akun login berhasil diperbarui!'); } else { return responseJSON('error', 'Gagal menemukan data akun yang lama.'); } } // --- PROSES TAMBAH DATA --- else if (action === 'add') { const timestamp = new Date().toLocaleString('id-ID'); db.sheetData.appendRow([timestamp, data.jenisAkun, data.namaPemilik, data.email, data.kataSandi]); return responseJSON('success', 'Data kredensial berhasil disimpan!'); } // --- PROSES EDIT DATA --- else if (action === 'edit') { const rowIndex = data.rowIndex; db.sheetData.getRange(rowIndex, 2, 1, 4).setValues([[data.jenisAkun, data.namaPemilik, data.email, data.kataSandi]]); return responseJSON('success', 'Data kredensial berhasil diperbarui!'); } // --- PROSES HAPUS DATA --- else if (action === 'delete') { const rowIndex = data.rowIndex; db.sheetData.deleteRow(rowIndex); return responseJSON('success', 'Data kredensial berhasil dihapus!'); } } catch (error) { return responseJSON('error', error.toString()); } } // API GET function doGet(e) { try { const db = initDatabase(); const data = db.sheetData.getDataRange().getValues(); const headers = data[0]; const result = []; const platformsSet = new Set(); if (data.length > 1) { for (let i = 1; i < data.length; i++) { let obj = { rowIndex: i + 1 }; for (let j = 0; j < headers.length; j++) { obj[headers[j]] = data[i][j]; } result.push(obj); if(data[i][1]) platformsSet.add(data[i][1].trim()); } } result.reverse(); return ContentService.createTextOutput(JSON.stringify({ status: 'success', data: result, platforms: Array.from(platformsSet) })).setMimeType(ContentService.MimeType.JSON); } catch (error) { return responseJSON('error', error.toString()); } } function responseJSON(status, message) { return ContentService.createTextOutput(JSON.stringify({ status: status, message: message })).setMimeType(ContentService.MimeType.JSON); }
